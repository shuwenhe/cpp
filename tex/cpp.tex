// for循环求和1+2+3+...+100
\begin{lstlisting}[language=C++]
int forSum(){
    int sum = 0;
    for (int i = 1; i <= 100; i++) {
        sum += i;
    }
    cout << "for sum=" << sum << endl;
    return 0;
}
\end{lstlisting}
